import React from 'react'

Required onChange handle to prevent react warning Actual handler is on parent element and triggered by bubbling

const nop = event => {} export const IndeterminateCheckbox = React.forwardRef( ({ indeterminate, checked, title, disabled }, ref) => { const defaultRef = React.useRef() const resolvedRef = ref || defaultRef React.useEffect(() => { resolvedRef.current.indeterminate = indeterminate }, [resolvedRef, indeterminate]) return ( <input type='checkbox' ref={resolvedRef} title={title} disabled={disabled} checked={checked} onChange={nop} /> ) } ) IndeterminateCheckbox.displayName = 'IndeterminateCheckbox'
